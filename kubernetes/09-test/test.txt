import requests
import time

def check_app_status(url):
    try:
        response = requests.get(url)
        # Check if the response status code is 2xx (indicating success)
        if response.status_code // 100 == 2:
            print(f"App is up and running - Status Code: {response.status_code}")
        else:
            print(f"App is down - Status Code: {response.status_code}")
    except requests.RequestException as e:
        print(f"Error connecting to the app: {e}")

if __name__ == "__main__":
    # Replace 'http://xyz.com' with your actual app URL
    app_url = 'http://xyz.com'

    # Run the check every minute
    while True:
        check_app_status(app_url)
        time.sleep(60)  # Sleep for 60 seconds before checking again
